import java.util.*;
class Main {
    public static void main(String[] args) {
        
       int number = 123;
        int no = reverseNum(number);
        System.out.println(no);
    }
    
    public static int reverseNum(int num)
    {
        int rev = 0;
        while(num != 0)
        {
            int digit = num % 10;
            rev = rev * 10 + digit;
            num = num / 10;
        }
        return rev;
    }
}
